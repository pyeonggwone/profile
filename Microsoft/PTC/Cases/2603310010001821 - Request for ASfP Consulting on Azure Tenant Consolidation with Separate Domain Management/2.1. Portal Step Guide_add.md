```env
TARGET_TENANT_ID=8f3c2a9e-5d41-4c7a-9b28-6e1f0d3a72c4
TARGET_PRIMARY_DOMAIN=kegroup.co.kr

PILOT_USER_GROUP=KEG Manufacturing Migration Pilot Users

PILOT_APPLICATION=Microsoft Office 365
CROSS_TENANT_SYNC_CONFIGURATION=Sync-KEG-Manufacturing-to-Group-Tenant-Pilot
```
